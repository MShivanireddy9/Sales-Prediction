import numpy as np
import xgboost as xgb
from sklearn.model_selection import train_test_split
from sklearn.metrics import mean_squared_error
import matplotlib.pyplot as plt


monthly_sales = data.set_index('Date')['Total Revenue (₹)'].resample('M').sum()

# Creating lag features (e.g., using previous 3 months to predict the next month)
def create_lag_features(series, n_lags):
    df = pd.DataFrame(series)
    for i in range(1, n_lags + 1):
        df[f'lag_{i}'] = series.shift(i)
    df = df.dropna()
    return df

# Prepare the data with 3 lag features
n_lags = 3
lagged_data = create_lag_features(monthly_sales, n_lags)

# Splitting into features (X) and target (y)
X = lagged_data.drop(columns=['Total Revenue (₹)'])
y = lagged_data['Total Revenue (₹)']

# Splitting into train and test sets
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, shuffle=False)

# Convert to XGBoost DMatrix format with feature names as a list of strings
train_dmatrix = xgb.DMatrix(X_train, label=y_train, feature_names=list(X_train.columns))
test_dmatrix = xgb.DMatrix(X_test, label=y_test, feature_names=list(X_test.columns))

# Train the XGBoost model
params = {
    'objective': 'reg:squarederror',
    'eval_metric': 'rmse',
    'seed': 42
}
model = xgb.train(params, train_dmatrix, num_boost_round=100, evals=[(train_dmatrix, 'train'), (test_dmatrix, 'eval')], early_stopping_rounds=10)

# Predict on the test set
y_pred = model.predict(test_dmatrix)

# Calculate the RMSE
rmse = np.sqrt(mean_squared_error(y_test, y_pred))
print(f"Test RMSE: {rmse}")

# Plot actual vs predicted values
plt.figure(figsize=(10, 6))
plt.plot(y_test.index, y_test, label='Actual Sales', color='blue')
plt.plot(y_test.index, y_pred, label='Predicted Sales', color='orange')
plt.title('XGBoost Sales Forecast')
plt.xlabel('Date')
plt.ylabel('Total Revenue (₹)')
plt.legend()
plt.show()

# Forecast future values (using the last 3 observations to predict the next 12 months)
future_forecast = []
last_obs = X_test.iloc[-1].values

for _ in range(12):
    # Create a DMatrix for prediction with the correct feature names
    future_dmatrix = xgb.DMatrix([last_obs], feature_names=list(X_test.columns))

    next_pred = model.predict(future_dmatrix)[0]
    future_forecast.append(next_pred)

    # Roll the observations and update the last lag with the predicted value
    last_obs = np.roll(last_obs, -1)
    last_obs[-1] = next_pred

# Plot future forecasted values
future_dates = pd.date_range(start=monthly_sales.index[-1] + pd.DateOffset(months=1), periods=12, freq='M')
plt.figure(figsize=(10, 6))
plt.plot(future_dates, future_forecast, label='XGBoost Forecasted Sales', color='green')
plt.title('XGBoost Future Sales Forecast')
plt.xlabel('Date')
plt.ylabel('Total Revenue (₹)')
plt.legend()
plt.show()

# Print future forecasted values
future_forecast_series = pd.Series(future_forecast, index=future_dates)
print(future_forecast_series)
